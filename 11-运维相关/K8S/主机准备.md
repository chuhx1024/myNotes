## 设置主机名称
```
hostnamectl set-hostname master1
hostnamectl set-hostname worker1
hostnamectl set-hostname worker2
```
## 设置IP地址
```
hostnamectl set-hostname master1  IPADDR="192.168.216.100"
hostnamectl set-hostname worker1  IPADDR="192.168.216.101"
hostnamectl set-hostname worker2  IPADDR="192.168.216.102"


vi /etc/sysconfig/network-scripts/ifcfg-ensss

BOOTPROTOT="null"
IPADDR="xxxxxxxx"
PREFIX="24"
GATEWAY="192.168.216.2"
DNSI="119.29.29.29"

# 重启网络

systemctl restart network

# 查看 设置
ip a s 

```
